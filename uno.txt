float number;
void setup()
{
  pinMode(13, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  if(Serial.available()>0){
    number=Serial.parseFloat();
    Serial.println(number);  }
  digitalWrite(13, HIGH);
  delay(1000/number); // Wait for 1000 millisecond(s)
  digitalWrite(13, LOW);
  delay(1000/number); // Wait for 1000 millisecond(s)
}
